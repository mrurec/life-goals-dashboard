Пример: Query resolver testExample: Query Resolver Test


@SpringBootTest
@DgsAutoConfiguration
class FitnessGraphQLTest {

    @Autowired
    private DgsQueryExecutor queryExecutor;

    @MockkBean
    private FitnessService fitnessService;

    @Test
    fun `myWorkouts query should return user workouts`() {
        // Arrange
        val userId = UUID.randomUUID()
        val workouts = listOf(
            Workout(id = UUID.randomUUID(), exercise = "Push-ups", sets = 3),
            Workout(id = UUID.randomUUID(), exercise = "Squats", sets = 4)
        )
        every { fitnessService.getWorkouts(userId) } returns workouts

        // Act
        val result = queryExecutor.executeAndExtractJsonPath<List<String>>(
            """
            {
                myWorkouts {
                    edges {
                        node {
                            id
                            exercise
                            sets
                        }
                    }
                }
            }
            """,
            "data.myWorkouts.edges[*].node.exercise"
        )

        // Assert
        assertThat(result).containsExactly("Push-ups", "Squats")
    }

    @Test
    fun `logWorkout mutation should create workout and return payload`() {
        every { fitnessService.logWorkout(any()) } returns workout

        val result = queryExecutor.execute(
            """
            mutation {
                logWorkout(input: {
                    exercise: "Bench Press"
                    sets: 5
                    reps: 10
                }) {
                    workout {
                        id
                        exercise
                    }
                    errors {
                        message
                    }
                }
            }
            """
        )

        assertThat(result.errors).isEmpty()
        assertThat(result.getData<Map<String, Any>>()
            .get("logWorkout")
            .get("workout")
            .get("exercise")
        ).isEqualTo("Bench Press")
    }
}

import { render, screen, fireEvent, waitFor } from '@testing-library/react';
import { MockedProvider } from '@apollo/client/testing';
import { createMockEnvironment, MockPayloadGenerator } from 'relay-test-utils';
import { WorkoutLogger } from './WorkoutLogger';

describe('WorkoutLogger', () => {
  let mockEnvironment;

  beforeEach(() => {
    mockEnvironment = createMockEnvironment();
  });

  test('should render form with inputs', () => {
    render(
      <MockedProvider mocks={[]}>
        <WorkoutLogger />
      </MockedProvider>
    );

    expect(screen.getByLabelText(/exercise/i)).toBeInTheDocument();
    expect(screen.getByLabelText(/sets/i)).toBeInTheDocument();
    expect(screen.getByRole('button', { name: /log/i })).toBeInTheDocument();
  });

  test('should submit form with valid data', async () => {
    render(
      <MockedProvider mocks={[]}>
        <WorkoutLogger />
      </MockedProvider>
    );

    fireEvent.change(screen.getByLabelText(/exercise/i), { target: { value: 'Squats' } });
    fireEvent.change(screen.getByLabelText(/sets/i), { target: { value: '4' } });
    fireEvent.click(screen.getByRole('button', { name: /log/i }));

    // Mock the GraphQL response
    mockEnvironment.mock.resolveMostRecentOperation(operation => {
      MockPayloadGenerator.generate(operation, {
        Workout: () => ({
          id: 'workout-1',
          exercise: 'Squats',
          sets: 4
        })
      });
    });

    await waitFor(() => {
      expect(screen.getByText(/workout logged/i)).toBeInTheDocument();
    });
  });

  test('should display error on submission failure', async () => {
    render(
      <MockedProvider mocks={[]}>
        <WorkoutLogger />
      </MockedProvider>
    );

    fireEvent.click(screen.getByRole('button', { name: /log/i }));

    mockEnvironment.mock.rejectMostRecentOperation(() => {
      return new Error('Failed to log workout');
    });

    await waitFor(() => {
      expect(screen.getByText(/error/i)).toBeInTheDocument();
    });
  });
});

// src/test/kotlin/com/mrurec/lifegoals/BaseIntegrationTest.kt
@SpringBootTest
@Testcontainers
abstract class BaseIntegrationTest {

    companion object {
        @Container
        val postgres = PostgreSQLContainer<Nothing>("postgres:16-alpine")
            .withDatabaseName("lifegoals_test")
            .withUsername("postgres")
            .withPassword("test")

        @Container
        val redis = GenericContainer<Nothing>("redis:7-alpine")
            .withExposedPorts(6379)

        @DynamicPropertySource
        @JvmStatic
        fun overrideProps(registry: DynamicPropertyRegistry) {
            registry.add("spring.datasource.url") { postgres.jdbcUrl }
            registry.add("spring.datasource.username") { postgres.username }
            registry.add("spring.datasource.password") { postgres.password }
            registry.add("spring.redis.host") { redis.host }
            registry.add("spring.redis.port") { redis.firstMappedPort }
        }
    }

    @Autowired protected lateinit var bookRepository: BookRepository
    @Autowired protected lateinit var redisTemplate: StringRedisTemplate

    @BeforeEach
    fun clearData() {
        bookRepository.deleteAll()
        redisTemplate.keys("*").forEach { key ->
            redisTemplate.delete(key)
        }
    }
}

// Конкретный тест наследует BaseIntegrationTest
class BookRepositoryIntegrationTest : BaseIntegrationTest() {

    @Test
    fun `should persist and retrieve book`() {
        val book = Book(
            userId = UUID.randomUUID(),
            title = "Clean Code",
            author = "Robert Martin",
            totalPages = 464
        )

        val saved = bookRepository.save(book)
        val retrieved = bookRepository.findById(saved.id).get()

        assertThat(retrieved.title).isEqualTo("Clean Code")
        assertThat(retrieved.currentPage).isEqualTo(0)
    }

    @Test
    fun `should enforce user_id scoping`() {
        val userId1 = UUID.randomUUID()
        val userId2 = UUID.randomUUID()

        val book1 = bookRepository.save(
            Book(userId = userId1, title = "Book 1", author = "A", totalPages = 100)
        )
        val book2 = bookRepository.save(
            Book(userId = userId2, title = "Book 2", author = "B", totalPages = 200)
        )

        val user1Books = bookRepository.findByUserId(userId1)
        assertThat(user1Books).hasSize(1)
        assertThat(user1Books[0].title).isEqualTo("Book 1")
    }

    @Test
    fun `should batch load books by IDs with viewer scoping`() {
        val userId = UUID.randomUUID()
        val books = (1..5).map { i ->
            bookRepository.save(
                Book(userId = userId, title = "Book $i", author = "A", totalPages = 100 * i)
            )
        }

        val idsToLoad = books.take(3).map { it.id }
        val loaded = bookRepository.findByUserIdAndIdIn(userId, idsToLoad)

        assertThat(loaded).hasSize(3)
        assertThat(loaded.map { it.title }).containsExactly("Book 1", "Book 2", "Book 3")
    }
}

@SpringBootTest
class ViewerContextSecurityTest {

    @MockkBean lateinit var bookRepository: BookRepository
    @Autowired lateinit var bookService: BookService
    @Autowired lateinit var viewerContextProvider: ViewerContextProvider

    @Test
    fun `service should not load other user's book`() {
        val ownerViewer = ViewerContext(userId = UUID.randomUUID(), email = "owner@test.com")
        val hackerViewer = ViewerContext(userId = UUID.randomUUID(), email = "hacker@test.com")

        val ownerBook = Book(
            userId = ownerViewer.userId,
            title = "Secret Book",
            author = "Author",
            totalPages = 100
        )

        every { bookRepository.findById(any()) } returns Optional.of(ownerBook)

        // Hacker tries to load owner's book
        val result = bookService.getBook(ownerBook.id, hackerViewer)

        // Should return NOT_FOUND (not PERMISSION_DENIED to avoid info leakage)
        assertThat(result).isNull()

        // Should NOT throw exception that exposes owner existence
    }

    @Test
    fun `repository query includes viewer user_id`() {
        val viewer = ViewerContext(userId = UUID.randomUUID(), email = "user@test.com")

        // Create a slot to capture the SQL query
        val queryCaptor = slot<Any>()
        every { bookRepository.findByUserId(capture(queryCaptor)) } returns emptyList()

        bookService.getUserBooks(viewer)

        // Verify the repository was called with viewer's ID
        verify { bookRepository.findByUserId(viewer.userId) }
    }

    @Test
    fun `DataLoader includes viewer user_id in batch query`() {
        val viewer = ViewerContext(userId = UUID.randomUUID(), email = "user@test.com")

        every { bookRepository.findByUserIdAndIdIn(any(), any()) } returns emptyList()

        // Simulate batch loading of 5 books
        val bookIds = (1..5).map { UUID.randomUUID() }
        bookService.loadBooksBatch(viewer, bookIds)

        // Verify batch query always includes viewer's userId
        verify {
            bookRepository.findByUserIdAndIdIn(
                viewer.userId,
                match { ids -> ids.size == 5 }
            )
        }
    }
}
